\section{Отрывок из учебника по математическому анализу \cite{math_analiz}}
\begin{center}
\textit{\S 3. Элементарные функции}
\end{center}
\hspace{0.5cm} Лемма 2. \textit{Пусть $P(x)$ и $Q(x)$ --- многочлены действительными коэффицентами, причем} $\frac{P(x)}{Q(x)}$ --- правильная рациональная дробь. \par
\indent \textit{Если число $z_0 = x_0 + y_0i$, $x_0 \in \mathbb{R}$, $y_0 \in \mathbb{R}$,
 $y_0 \neq 0$, является корнем кратности $m \geq 1$ многочлена $Q(x)$, т. е.}
\begin{equation}  
    \label{eq:1}
    Q(x) = (x^2+px+q)^mQ_1(x)
\end{equation}
\textit{где}

\begin{equation}  
    \label{eq:2}
    Q_1(z_0) \neq 0,\quad x^2 +px + q = (x-z_0)(x-\bar{z_0})
\end{equation}

\textit{то существуют такие единственные действительные числа $B$, $C$
и многочлен $P_1(x)$ с действительными коэфицентами, что }

\begin{equation}
    \label{eq:3}
    \frac{P(x)}{Q(x)}=\frac{B_x+C}{(x^2+px+q)^m}+\frac{P_1(X)}{(x^2+px+q)^{m-1}Q_1(x)}
\end{equation}

\textit{где дробь}
\begin{equation}
    \label{eq:4}
    \frac{P_1(x)}{(x^2+px+q)^{m-1}Q_1(m)}
\end{equation}
\textit{также является правильной.} \par

\noindent $\triangleright$ Для любых действительных чисел $B$ и $C$ имеем \par
\noindent $\frac{P(x)}{Q(x)}=\frac{P(x)}{(x^2+px+q)^mQ_1(x)}=$ \par
$=\frac{B_x+C}{(x^2+px+q)^m} + 
\Biggl[\frac{P(x)}{(x^2+px+q)^(m)Q_1(x)} - \frac{B_x+C}{(x^2+px+q)^m} \Biggr] =$ \par
\hspace{2.5cm} $= \frac{B_x+C}{(x^2+px+q)^m}+\frac{P(x)-(B_x+C)Q_1(x)}{(x^2+px+q)^mQ_1(x)}$ 
\par

\indent \textit{Рассуждениями, аналогичными проведенным при доказательстве леммы 1, легко убедиться, что дробь}
\begin{equation}
    \label{eq:5}
    \frac{P(x)-(B_x+C)Q_1(x)}{(x^2+px+q)^mQ_1(x)}
\end{equation}
является правильной и что коэфиценты многочленов, стоящих у нее в числителе и знаменателе, являются действительными. \par
\indent Для того чтобы дробь (\ref{eq:5}) имела вид
\begin{equation}
    \label{eq:6}
    \frac{P_1(x)}{(x^2+px+q)^{m-1}Q_1(x)}
\end{equation}
необходимо и достаточно, чтобы ее множитель $x^2+px+q=(x-z_0)(x-\bar{z_0})$, а это можно сделать тогда и только тогда, когда числитель дроби (\ref{eq:5}) делится на $x^2+px+q=(x-z_0)(x-\bar{z_0})$,
что согласно теореме Безу равносильно тому, что число $z_0$ является норнем многочлена, стоящего в числители дроби (\ref{eq:5}), т. е. когда
\[P(z_0)-(Bz_0+C)Q_1(z_0)=0.\]
